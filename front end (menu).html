const express = require('express'); const app = express(); const port = 3000; // Sample menu data (in a real-world scenario, you'd fetch this from a database) const menu = [ { id: 1, name: 'Chicken Burger', description: 'Grilled chicken patty with lettuce, tomato, and mayonnaise.', price: 8.99 }, { id: 2, name: 'Margherita Pizza', description: 'Fresh tomatoes, mozzarella, and basil on a thin crust.', price: 12.99 }, { id: 3, name: 'Caesar Salad', description: 'Romaine lettuce, croutons, Parmesan cheese, and Caesar dressing.', price: 6.99 } ]; // Set up a route to get the menu app.get('/menu', (req, res) => { res.json(menu); }); // Set up a route to get a specific menu item by ID app.get('/menu/:id', (req, res) => { const menuItem = menu.find(item => item.id === parseInt(req.params.id)); if (!menuItem) { res.status(404).json({ error: 'Menu item not found' }); } else { res.json(menuItem); } }); app.listen(port, () => { console.log(`Server is running at http://localhost:${port}`); });
